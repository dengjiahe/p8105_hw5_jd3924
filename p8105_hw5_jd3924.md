sim_funct = function(
    n = 30, 
    mu = 0,
    sigma = 5) {
  x = rnorm(n, mean = mu, sd = sigma)
  t_test = t.test(x, conf.int = 0.95) %>% broom::tidy()
  
  t_test
}
output = vector("list",5000)
for (i in 1: 5000) {
  output[[i]] = sim_funct()
}
output %>% bind_rows() %>% head()
